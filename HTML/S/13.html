#include "population.h"
#include <QDebug>
Population::Population(int num)
  :gene_max(num),
   IndivNum(9),
   LossNum(6)
{
  for(int i = 0;i<9;i++)
    {
      Individual tmp(num);
      indiv.push_back(tmp);
    }
}
QVector<Individual>::iterator Population::begin()
{
  return indiv.begin();
}
Individual& Population::at(int i)
{
  return indiv[i];
}
